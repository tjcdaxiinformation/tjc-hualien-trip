        /**

         * 報名統計邏輯說明：

         * 1. 如果您沒有 API，可以手動修改 manualCount 變數。

         * 2. 若要串接 Google Sheets，請參考註解掉的 fetch 部分。

         */

        document.addEventListener('DOMContentLoaded', () => {

            const totalCapacity = 84;

            const currentElement = document.getElementById('current-count');

            const remainingElement = document.getElementById('remaining-count');

            const progressFill = document.getElementById('progress-fill');

            const statusText = document.getElementById('status-text');


            // --- 手動更新模式 (建議方案) ---

            let manualCount = 65; // <--- 每次有人報名，只需修改 GitHub 上的這個數字並存檔

            

            updateDisplay(manualCount);


            function updateDisplay(count) {

                currentElement.innerText = count;

                const remaining = totalCapacity - count;

                const percentage = (count / totalCapacity) * 100;

                

                progressFill.style.width = percentage + '%';

                statusText.innerHTML = `感謝主！目前僅剩 <span class="text-red-500 font-bold">${remaining}</span> 個名額，欲報從速！`;

                

                if (count >= totalCapacity) {

                    statusText.innerHTML = "✨ 感謝主！名額已滿，進入候補名單。";

                    statusText.classList.add("text-emerald-600");

                }

            }


            /* // --- 自動更新模式 (進階：串接 Google Sheets CSV) ---

            // 需先在試算表點擊「發布到網路」並選擇 CSV 格式

            const CSV_URL = '您的_GOOGLE_SHEET_CSV_發布網址';

            

            fetch(CSV_URL)

                .then(res => res.text())

                .then(data => {

                    // 假設試算表第一行之後每行代表一人

                    const rows = data.split('\n').length - 1; 

                    updateDisplay(rows);

                })

                .catch(err => {

                    console.error("無法讀取統計數據", err);

                    updateDisplay(manualCount); // 出錯時使用手動數據

                });

            */

        });
